## 🚀 Quick Start

```tsx
import { DatePicker } from 'calendrax'
import 'calendrax/styles.css'

function App() {
  const [dates, setDates] = useState({ checkin: null, checkout: null })
  const [open, setOpen] = useState(false)

  return (
    <DatePicker
      dates={dates}
      setDates={setDates}
      open={open}
      setOpen={setOpen}
      events={events}
    />
  )
}
